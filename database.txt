create database scandiwebTestDb;

use scandiwebTestDb;

create table productType(
    id int(11) not null auto_increment primary key,
    type varchar(10) not null
);

create table product(
    id int(11) not null auto_increment primary key,
    sku varchar(20) not null,
    name varchar(20) not null,
    price integer(11) not null,
    type integer(11),
    constraint sku_unique unique(sku),
    constraint prod_type_id foreign key(type) references productType(id) on delete cascade
);

create table productDescription(
    id int(11) not null auto_increment primary key,
    description json default null,
    product integer(11) not null,
    constraint opt_prod_id foreign key(product) references product(id) on delete cascade
);
